./configure: configure pcb-rnd.

Usage: ./configure [options]

options are:
 --prefix=path              change installation prefix from /usr to path
 --debug                    build full debug version (-g -O0, extra asserts)
 --symbols                  include symbols (add -g, but no -O0 or extra asserts)
 --coord=32|64              set coordinate integer type's width in bits
 --dot_pcb_pcb=path         .pcb-rnd config path under $HOME/
 --workaround-gtk-ctrl      enable GTK control key query workaround
 --all=plugin               enable all working plugins for dynamic load
 --all=buildin              enable all working plugins for static link
 --all=disable              disable all plugins (compile core only)
 --force-all=plugin         enable even broken plugins for dynamic load
 --force-all=buildin        enable even broken plugins for static link
 --disable-xrender          do not use xrender for lesstif
